// ================================================================ // IURECO — Web App v2 // Pegar en: Google Sheets → Extensiones → Apps Script // Publicar como: Web App → Cualquier persona // ================================================================ const SHEET_CONFIG = { FORM_RESPONSES: 'Respuestas de formulario 1', CONTROL_SHEET: 'Control', PAGOS_SHEET: 'Pagos', GASTOS_SHEET: 'Gastos', TICKET_PREFIX: 'IUR', }; const REQUISITOS = [ 'Tres nombres tentativos de la sociedad', 'Nombre tentativo empresa mercantil', 'Copia DPI accionistas', 'Profesión u oficio de cada accionista', 'Copia DPI representante legal', 'Distribución de acciones (%)', 'Objeto de la sociedad', 'Copia recibo agua o luz', 'Nombre y NIT del Contador', 'Régimen tributario', 'Capital social a autorizar (Q)', 'Capital suscrito y pagado (Q)', 'Boleta de anticipo' ]; const PASOS_TRAMITE = [ 'Redacción de borrador de escritura', 'Impresión y firma de escritura pública', 'Preparación de documentación', 'Subida a Registro Mercantil', 'Seguimiento del proceso', 'Descarga de patentes', 'Entrega de documentación' ]; // ================================================================ // ROUTER // ================================================================ function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const params = e.parameter || {}; const action = params.action || ''; const body = {}; Object.keys(params).forEach(k => { try { body[k] = JSON.parse(params[k]); } catch(err) { body[k] = params[k]; } }); let result; try { switch (action) { case 'getExpedientes': result = getExpedientes(); break; case 'getExpediente': result = getExpediente(body.ticket); break; case 'updateEstatus': result = updateEstatus(body); break; case 'updatePaso': result = updatePaso(body); break; case 'updateNotas': result = updateNotas(body); break; case 'marcarFaltantes': result = marcarFaltantes(body); break; case 'generarFormCorr': result = generarFormCorreccion(body); break; case 'agregarWhatsApp': result = agregarWhatsApp(body); break; case 'agregarPago': result = agregarPago(body); break; case 'confirmarEntrega': result = confirmarEntrega(body); break; case 'getPagos': result = getPagos(body.ticket); break; case 'agregarGasto': result = agregarGasto(body); break; case 'getGastos': result = getGastos(body.ticket); break; case 'getResumenFinanciero': result = getResumenFinanciero(body.ticket); break; default: result = { error: 'Acción no reconocida: ' + action }; } } catch (err) { result = { error: err.message }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // ================================================================ // HELPERS // ================================================================ function getSSS() { return SpreadsheetApp.getActiveSpreadsheet(); } function updateField(ticket, campo, valor) { const ss = getSSS(); const ctrl = ss.getSheetByName(SHEET_CONFIG.CONTROL_SHEET); const data = ctrl.getDataRange().getValues(); const headers = data[0]; const col = headers.indexOf(campo); if (col < 0) return { error: 'Campo no encontrado: ' + campo }; for (let i = 1; i < data.length; i++) { if (data[i][0] === ticket) { ctrl.getRange(i + 1, col + 1).setValue(valor); return { ok: true }; } } return { error: 'Ticket no encontrado' }; } function getRowByTicket(sheetName, ticket) { const ss = getSSS(); const sheet = ss.getSheetByName(sheetName); if (!sheet) return null; const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === ticket) return { row: i + 1, data: data[i], headers: data[0] }; } return null; } // ================================================================ // INICIALIZAR HOJAS // ================================================================ function asegurarHojas() { const ss = getSSS(); // Hoja Control let ctrl = ss.getSheetByName(SHEET_CONFIG.CONTROL_SHEET); if (!ctrl) { ctrl = ss.insertSheet(SHEET_CONFIG.CONTROL_SHEET); const headers = [ 'Ticket','Cliente','WhatsApp','Fecha Ingreso','Estatus', 'Paso Actual','Anticipo Confirmado','Entrega Confirmada','Notas','Link Form Corrección' ].concat(REQUISITOS.map(r => 'FALTA: ' + r)); ctrl.getRange(1,1,1,headers.length).setValues([headers]); ctrl.getRange(1,1,1,headers.length).setBackground('#1A2744').setFontColor('#FFFFFF').setFontWeight('bold'); ctrl.setFrozenRows(1); } // Hoja Pagos let pagos = ss.getSheetByName(SHEET_CONFIG.PAGOS_SHEET); if (!pagos) { pagos = ss.insertSheet(SHEET_CONFIG.PAGOS_SHEET); pagos.getRange(1,1,1,5).setValues([['Ticket','Cliente','Tipo','Monto (Q)','Fecha']]); pagos.getRange(1,1,1,5).setBackground('#1A2744').setFontColor('#FFFFFF').setFontWeight('bold'); pagos.setFrozenRows(1); } // Hoja Gastos let gastos = ss.getSheetByName(SHEET_CONFIG.GASTOS_SHEET); if (!gastos) { gastos = ss.insertSheet(SHEET_CONFIG.GASTOS_SHEET); gastos.getRange(1,1,1,5).setValues([['Ticket','Cliente','Descripción','Monto (Q)','Fecha']]); gastos.getRange(1,1,1,5).setBackground('#1A2744').setFontColor('#FFFFFF').setFontWeight('bold'); gastos.setFrozenRows(1); } return { ctrl, pagos, gastos }; } // ================================================================ // EXPEDIENTES // ================================================================ function getExpedientes() { const ss = getSSS(); const ctrl = ss.getSheetByName(SHEET_CONFIG.CONTROL_SHEET); if (!ctrl) return { expedientes: [] }; const data = ctrl.getDataRange().getValues(); const headers = data[0]; const expedientes = data.slice(1) .filter(r => r[0] !== '') .map(r => { const obj = {}; headers.forEach((h, i) => obj[h] = r[i]); return obj; }); return { expedientes }; } function getExpediente(ticket) { const ss = getSSS(); const ctrl = ss.getSheetByName(SHEET_CONFIG.CONTROL_SHEET); const data = ctrl.getDataRange().getValues(); const headers = data[0]; for (let i = 1; i < data.length; i++) { if (data[i][0] === ticket) { const obj = {}; headers.forEach((h, idx) => obj[h] = data[i][idx]); return { expediente: obj, row: i + 1 }; } } return { error: 'Ticket no encontrado' }; } function updateEstatus(body) { return updateField(body.ticket, 'Estatus', body.estatus); } function updatePaso(body) { return updateField(body.ticket, 'Paso Actual', body.paso); } function updateNotas(body) { return updateField(body.ticket, 'Notas', body.notas); } function agregarWhatsApp(body) { return updateField(body.ticket, 'WhatsApp', body.whatsapp); } // ================================================================ // PAGOS // ================================================================ function agregarPago(body) { const { ticket, tipo, monto } = body; // tipo: 'Anticipo' | 'Contra entrega' const ss = getSSS(); const { pagos } = asegurarHojas(); // Buscar nombre del cliente const expData = getExpediente(ticket); const cliente = expData.expediente ? expData.expediente['Cliente'] : ''; pagos.appendRow([ticket, cliente, tipo, parseFloat(monto) || 0, new Date().toLocaleDateString('es-GT')]); // Si es anticipo, marcar anticipo confirmado en Control if (tipo === 'Anticipo') { updateField(ticket, 'Anticipo Confirmado', 'Sí'); updateField(ticket, 'Estatus', 'Expediente en operación'); } return { ok: true }; } function confirmarEntrega(body) { const { ticket, monto } = body; const ss = getSSS(); const { pagos } = asegurarHojas(); const expData = getExpediente(ticket); const cliente = expData.expediente ? expData.expediente['Cliente'] : ''; pagos.appendRow([ticket, cliente, 'Contra entrega', parseFloat(monto) || 0, new Date().toLocaleDateString('es-GT')]); updateField(ticket, 'Entrega Confirmada', 'Sí'); updateField(ticket, 'Estatus', 'Entregado'); updateField(ticket, 'Paso Actual', 'Entrega de documentación'); return { ok: true }; } function getPagos(ticket) { const ss = getSSS(); const pagos = ss.getSheetByName(SHEET_CONFIG.PAGOS_SHEET); if (!pagos) return { pagos: [] }; const data = pagos.getDataRange().getValues(); const headers = data[0]; const filas = data.slice(1) .filter(r => r[0] === ticket) .map(r => { const obj = {}; headers.forEach((h, i) => obj[h] = r[i]); return obj; }); return { pagos: filas }; } // ================================================================ // GASTOS // ================================================================ function agregarGasto(body) { const { ticket, descripcion, monto } = body; const ss = getSSS(); const { gastos } = asegurarHojas(); const expData = getExpediente(ticket); const cliente = expData.expediente ? expData.expediente['Cliente'] : ''; gastos.appendRow([ticket, cliente, descripcion, parseFloat(monto) || 0, new Date().toLocaleDateString('es-GT')]); return { ok: true }; } function getGastos(ticket) { const ss = getSSS(); const gastos = ss.getSheetByName(SHEET_CONFIG.GASTOS_SHEET); if (!gastos) return { gastos: [] }; const data = gastos.getDataRange().getValues(); const headers = data[0]; const filas = data.slice(1) .filter(r => r[0] === ticket) .map(r => { const obj = {}; headers.forEach((h, i) => obj[h] = r[i]); return obj; }); return { gastos: filas }; } // ================================================================ // RESUMEN FINANCIERO // ================================================================ function getResumenFinanciero(ticket) { const pagosData = getPagos(ticket).pagos; const gastosData = getGastos(ticket).gastos; const totalAnticipo = pagosData.filter(p => p['Tipo'] === 'Anticipo').reduce((s, p) => s + (parseFloat(p['Monto (Q)']) || 0), 0); const totalEntrega = pagosData.filter(p => p['Tipo'] === 'Contra entrega').reduce((s, p) => s + (parseFloat(p['Monto (Q)']) || 0), 0); const totalIngresos = totalAnticipo + totalEntrega; const totalGastos = gastosData.reduce((s, g) => s + (parseFloat(g['Monto (Q)']) || 0), 0); const gananciaNeta = totalIngresos - totalGastos; return { totalAnticipo, totalEntrega, totalIngresos, totalGastos, gananciaNeta, pagos: pagosData, gastos: gastosData }; } // ================================================================ // DOCUMENTOS FALTANTES Y FORM DE CORRECCIÓN // ================================================================ function marcarFaltantes(body) { const { ticket, faltantes } = body; const ss = getSSS(); const ctrl = ss.getSheetByName(SHEET_CONFIG.CONTROL_SHEET); const data = ctrl.getDataRange().getValues(); const headers = data[0]; for (let i = 1; i < data.length; i++) { if (data[i][0] === ticket) { REQUISITOS.forEach(req => { const col = headers.indexOf('FALTA: ' + req); if (col >= 0) ctrl.getRange(i+1, col+1).setValue(Array.isArray(faltantes) && faltantes.includes(req)); }); return { ok: true }; } } return { error: 'Ticket no encontrado' }; } function generarFormCorreccion(body) { const { ticket } = body; const ss = getSSS(); const ctrl = ss.getSheetByName(SHEET_CONFIG.CONTROL_SHEET); const data = ctrl.getDataRange().getValues(); const headers = data[0]; let filaCliente = null, rowIndex = -1; for (let i = 1; i < data.length; i++) { if (data[i][0] === ticket) { filaCliente = data[i]; rowIndex = i + 1; break; } } if (!filaCliente) return { error: 'Ticket no encontrado' }; const faltantes = REQUISITOS.filter(req => { const col = headers.indexOf('FALTA: ' + req); return col >= 0 && (filaCliente[col] === true || filaCliente[col] === 'TRUE'); }); if (!faltantes.length) return { error: 'No hay requisitos marcados como faltantes' }; const clienteNombre = filaCliente[1]; const form = FormApp.create(`Corrección — ${ticket}`); form.setTitle(`Corrección de documentos — ${ticket}`); form.setDescription(`Estimado/a ${clienteNombre},\n\nNecesitamos la siguiente documentación faltante.\n\nExpediente: ${ticket}`); form.setConfirmationMessage(`Gracias. Corrección recibida. Expediente: ${ticket}`); form.addTextItem().setTitle('Número de expediente').setRequired(true); const TIPO = { 'Copia DPI accionistas':'archivo','Copia DPI representante legal':'archivo','Copia recibo agua o luz':'archivo','Boleta de anticipo':'archivo' }; faltantes.forEach(req => { if (TIPO[req] === 'archivo') form.addFileUploadItem().setTitle(req).setRequired(true); else form.addParagraphTextItem().setTitle(req).setRequired(true); }); const formUrl = form.getPublishedUrl(); const linkCol = headers.indexOf('Link Form Corrección'); if (linkCol >= 0) ctrl.getRange(rowIndex, linkCol+1).setValue(formUrl); updateField(ticket, 'Estatus', 'Corrección solicitada'); return { ok: true, formUrl, faltantes, cliente: clienteNombre }; } // ================================================================ // TRIGGER AL ENVIAR FORMULARIO // ================================================================ function onFormSubmit(e) { const ss = getSSS(); const sheet = ss.getSheetByName(SHEET_CONFIG.FORM_RESPONSES) || ss.getActiveSheet(); const lastRow = sheet.getLastRow(); const year = new Date().getFullYear(); const num = String(lastRow - 1).padStart(3, '0'); const ticketId = `${SHEET_CONFIG.TICKET_PREFIX}-${year}-${num}`; sheet.getRange(lastRow, sheet.getLastColumn()+1).setValue(ticketId).setBackground('#D9EAD3'); asegurarHojas(); const ctrl = ss.getSheetByName(SHEET_CONFIG.CONTROL_SHEET); const values = e.values; const nombre = values[1] || 'Sin nombre'; const fecha = new Date().toLocaleDateString('es-GT'); const fila = [ticketId, nombre, '', fecha, 'Pendiente revisión', '', 'No', 'No', '', '']; REQUISITOS.forEach(() => fila.push(false)); ctrl.appendRow(fila); const newRow = ctrl.getLastRow(); ctrl.getRange(newRow, 11, 1, REQUISITOS.length).insertCheckboxes(); try { const form = FormApp.openByUrl(ss.getFormUrl()); form.setConfirmationMessage(`✅ Documentación recibida.\n\nTu número de expediente es:\n${ticketId}\n\nAnótalo y envíalo al grupo de WhatsApp de IureCo.`); } catch(err) { Logger.log('No se pudo actualizar confirmación: ' + err); } } function onOpen() { SpreadsheetApp.getUi().createMenu('IureCo') .addItem('Inicializar hojas', 'asegurarHojas') .addToUi(); }
